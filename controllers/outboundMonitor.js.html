import { Op } from 'sequelize';
import moment from 'moment';
 
import OutboundRangeInformation from '../source/outboundRange/information';
import OutboundRangeUpdate from '../source/outboundRange/update';
import OutboundOrderUpdate from '../source/outboundOrders/update';
import UserInformation from '../source/users/information';
 
async function getDataGrid(req, res) {
  const { startDate, endDate, type } = req.body;
  const { userId } = req;
 
  try {
    const startDateF = moment(startDate).startOf('day').format();
    const endDateF = moment(endDate).endOf('day').format();
 
    let statusRange = ['99'];
    let statusOrder = ['99'];
    let statusRangeProd = ['99'];
 
    switch (type) {
      case 'pendingCut':
        statusRange = ['1', '2'];
        statusOrder = null;
        statusRangeProd = ['1'];
        break;
      case 'pendingPicking':
        statusRange = ['1'];
        statusOrder = null;
        statusRangeProd = ['2', '3'];
        break;
      case 'picking':
        statusRange = ['2'];
        statusOrder = null;
        statusRangeProd = ['2', '3'];
        break;
      case 'divergence':
        statusRange = ['5'];
        statusOrder = null;
        statusRangeProd = null;
        break;
      case 'picked':
        statusRange = ['2', '3'];
        statusOrder = ['2', '3', '4', '5'];
        statusRangeProd = ['4'];
        break;
      case 'check':
        statusRange = ['4'];
        statusOrder = ['5'];
        statusRangeProd = ['5'];
        break;
      case 'dockControl':
        statusRange = ['4'];
        statusOrder = ['6'];
        statusRangeProd = null;
        break;
      case 'consolidation':
        statusRange = null;
        statusOrder = ['8'];
        statusRangeProd = null;
        break;
      case 'packingList':
        statusRange = ['4'];
        statusOrder = ['7'];
        statusRangeProd = null;
        break;
      case 'withdrawStore':
        statusRange = null;
        statusOrder = ['9'];
        statusRangeProd = null;
        break;
      default:
        break;
    }
 
    const user = await UserInformation.getUser({
      id: userId,
    });
 
    let queryRange = {
      branchCode: user ? user.mainBranch : '',
      createdAt: {
        [Op.between]: [startDateF, endDateF],
      },
    };
 
    if (statusRange) {
      queryRange = {
        status: statusRange,
        ...queryRange,
      };
    }
 
    let queryOrder = {};
 
    if (statusOrder) {
      queryOrder = {
        status: statusOrder,
        restrictionPicking: false,
      };
    } else {
      queryOrder = {
        restrictionPicking: false,
      };
    }
 
    let queryProd = {};
 
    if (statusRangeProd) {
      queryProd = {
        status: statusRangeProd,
      };
    }
 
    const ordersDb = await OutboundRangeInformation.getOutboundRangeList(
      queryRange,
      queryOrder,
      queryProd
    );
 
    const rows = ordersDb.rows.map((o) => o.get());
 
    rows.forEach((row) => {
      const { restrictionPicking } = row.outboundOrder;
      if (restrictionPicking === false) {
        row.branchName = row.branch
          ? `${row.branch.code} - ${row.branch.name}`
          : row.branchCode;
        row.carrierName =
          row.outboundOrder && row.outboundOrder.carrier
            ? `${row.outboundOrder.carrier.code} - ${row.outboundOrder.carrier.name}`
            : row.outboundOrder.carrierCode;
        row.customerName =
          row.outboundOrder && row.outboundOrder.customer
            ? `${row.outboundOrder.customer.code} - ${row.outboundOrder.customer.name}`
            : row.outboundOrder.customerCode;
      }
    });
 
    res.json({
      success: true,
      data: rows,
    });
  } catch (err) {
    // eslint-disable-next-line no-console
    console.log(err);
    const response = {
      success: false,
    };
    res.json(response);
  }
}

async function allocatedProducts(req, res) {
  const { outboundRangeId } = req.query;
  try {
    const outboundRangeProducts =
      await OutboundRangeInformation.getOutboundRangeProducts({
        outboundRangeId,
      });
    const allocation = [];
 
    for (let index = 0; index < outboundRangeProducts.length; index += 1) {
      const element = outboundRangeProducts[index];
      allocation.push({
        branchCode:
          element.outboundOrder &&
          element.outboundOrder.branch &&
          element.outboundOrder.branch.code
            ? element.outboundOrder.branch.code
            : '',
        branchName:
          element.outboundOrder &&
          element.outboundOrder.branch &&
          element.outboundOrder.branch.name
            ? element.outboundOrder.branch.name
            : '',
        productCode:
          element.product && element.product.productCode
            ? element.product.productCode
            : '',
        productName:
          element.product && element.product.name ? element.product.name : '',
        customerName:
          element.outboundOrder &&
          element.outboundOrder.customer &&
          element.outboundOrder.customer.name
            ? element.outboundOrder.customer.name
            : '',
        orderNumber:
          element.outboundOrder && element.outboundOrder.orderNumber
            ? element.outboundOrder.orderNumber
            : '',
        customerOrderCode:
          element.outboundOrder && element.outboundOrder.customerOrderCode
            ? element.outboundOrder.customerOrderCode
            : '',
        deliveryDate:
          element.outboundOrder && element.outboundOrder.deliveryDate
            ? element.outboundOrder.deliveryDate
            : '',
        quantity: element.quantity ? element.quantity : '',
        addressCode:
          element.allocation &&
          element.allocation.balance &&
          element.allocation.balance.addressCode
            ? element.allocation.balance.addressCode
            : '',
        lotNumber:
          element.allocation &&
          element.allocation.balance &&
          element.allocation.balance.lotNumber
            ? element.allocation.balance.lotNumber
            : '',
        expirationDate:
          element.allocation &&
          element.allocation.balance &&
          element.allocation.balance.expirationDate
            ? element.allocation.balance.expirationDate
            : '',
      });
    }
    res.json({ success: true, data: allocation });
  } catch (err) {
    // eslint-disable-next-line no-console
    console.log(err);
    const response = {
      success: false,
    };
    res.json(response);
  }
}
 
async function updateOutboundRange(req, res) {
  const { outboundRangeId, fieldsOutboundRange } = req.body;
  const outboundRange = {
    ...fieldsOutboundRange,
    updatedUser: req.userId,
  };
  Object.keys(outboundRange).forEach(
    (key) =>
      key !== 'note' && outboundRange[key] === null && delete outboundRange[key]
  );
  try {
    await OutboundRangeUpdate.updateOutboundRange(
      outboundRangeId,
      outboundRange
    );
    res.json({ success: true });
  } catch (err) {
    // eslint-disable-next-line no-console
    console.log(err);
    const response = {
      success: false,
    };
    res.json(response);
  }
}
