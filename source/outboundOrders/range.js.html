async function createOutboundRanges(
  outboundOrderId,
  allocationsParam,
  userId,
  transaction,
  outboundOrderNote = '',
  industry = false,
  sequence = 1,
  pickedUser = null,
  outboundWaveId = null
) {
  try {
    const outboundOrder = await OutboundOrderInformation.getOutboundOrder(
      {
        id: outboundOrderId,
      },
      { transaction }
    );
    let allocations = allocationsParam || [];
 
    const branchInfo = await BranchInformation.getBranchSettings(
      {
        code: outboundOrder.branchCode,
      },
      { transaction }
    );
 
    if (!allocations || allocations.length === 0) {
      allocations =
        await ProductAllocationInformation.getOutboundOrderProductAllocations(
          {
            outboundOrderId: outboundOrder.id,
            status: [1, 3],
            sequence,
          },
          { transaction }
        );
    }
 
    for (const allocation of allocations) {
      const stockBalance = await StockInformation.getStockBalance(
        {
          id: allocation.stockBalanceId,
        },
        { transaction }
      );
 
      if (stockBalance) {
        const doNotValidateRange = !!(industry && !stockBalance.addressCode);
 
        const address = !doNotValidateRange
          ? await StorageAddressInformation.getStorageAddress(
              {
                branchCode: stockBalance.branchCode,
                code: stockBalance.addressCode,
              },
              { transaction }
            )
          : null;
 
        if (!address && !doNotValidateRange) {
          transaction.rollback();
          throw new APIError('', t('BEE3180' /* Endereço inválido! */));
        }
 
        const range = !doNotValidateRange
          ? await RangeInformation.getRange(
              {
                branchCode: address.branchCode,
                code: address.rangeCode,
              },
              { transaction }
            )
          : null;
 
        if (!range && !doNotValidateRange) {
          // transaction.rollback();
          throw new APIError('', t('BEE3181' /* Endereço não possui Range! */));
        }
 
        const rangeCode = doNotValidateRange
          ? outboundOrder.depositCode
          : range.code;
 
        let outboundOrderRange =
          await OutboundRangeInformation.getOutboundRange(
            {
              outboundOrderId: outboundOrder.id,
              rangeCode,
              sequence,
            },
            { transaction }
          );
 
        let userWithRangePermission = false;
        if (pickedUser) {
          const UserRange = await UserRangeInformation.getUsersRanges(
            {
              userId: pickedUser,
              rangeCode,
            },
            { transaction }
          );
 
          if (UserRange) {
            userWithRangePermission = true;
          }
        }
 
        if (!outboundOrderRange) {
          outboundOrderRange = await OutboundRangeCreate.createOutboundRange(
            {
              outboundOrderId: outboundOrder.id,
              rangeCode,
              priority: outboundOrder.priority,
              branchCode: outboundOrder.branchCode,
              sequence,
              status: industry ? 2 : 1,
              note: outboundOrderNote,
              createdUser: userId,
              updatedUser: userId,
              ...(industry
                ? {
                    pickedUser: userId,
                    pickedAt: moment().format(),
                    checkedUser: userId,
                    checkedAt: moment().format(),
                  }
                : {}),
              ...(userWithRangePermission ? { pickedUser } : {}),
              ...(userWithRangePermission && outboundWaveId
                ? { outboundWaveId }
                : {}),
            },
            {
              transaction,
            }
          );
 
          if (outboundWaveId) {
            await OutboundWavesUpdate.updateOutboundWave(
              outboundWaveId,
              {
                status: 2, // Em Separação
                ...(userWithRangePermission ? { pickedUser } : {}),
              },
              {
                transaction,
              }
            );
          }
        } else if (![1, 2].includes(parseInt(outboundOrderRange.status, 10))) {
          // Diferente de Pendente Separação, Em Separação
          await OutboundRangeUpdate.updateOutboundRange(
            outboundOrderRange.id,
            {
              status: 2, // Em Separação,
              ...(userWithRangePermission ? { pickedUser } : {}),
              ...(userWithRangePermission && outboundWaveId
                ? { outboundWaveId }
                : {}),
            },
            { transaction }
          );
 
          if (outboundWaveId || outboundOrderRange.outboundWaveId) {
            await OutboundWavesUpdate.updateOutboundWave(
              outboundWaveId || outboundOrderRange.outboundWaveId,
              {
                status: 2, // Em Separação
                ...(userWithRangePermission ? { pickedUser } : {}),
              },
              {
                transaction,
              }
            );
          }
        }
 
        const product = await ProductInformation.getProduct(
          {
            productCode: stockBalance.productCode,
          },
          { transaction }
        );
 
        let status;
 
        if (industry) {
          status = 5; // Finalizado
        } else {
          status =
            (outboundOrder.typeOrders
              ? outboundOrder.typeOrders.usesCableCuttingPlan
              : branchInfo.usesCableCuttingPlan) &&
            product.family.cuttingPlan &&
            product.stockControlType === 3
              ? 1 // Pendente Corte de Cabos
              : 2; // Pendente Separação
        }
 
        const existRangeProduct =
          await OutboundRangeInformation.getOutboundRangeProduct(
            {
              outboundOrderProductId: allocation.outboundOrderProductId,
              allocationId: allocation.id,
            },
            { transaction }
          );
 
        if (existRangeProduct) {
          await OutboundRangeUpdate.updateOutboundRangeProduct(
            existRangeProduct.id,
            {
              quantity: allocation.quantity,
              status,
              cuttingPlan: status === 1,
              updatedUser: userId,
              picked: false,
              ...(industry
                ? {
                    storageAddressPicked: stockBalance.addressCode,
                    pickedQuantity: allocation.quantity,
                    pickedAt: moment().format(),
                    pickedUser: userId,
                    picked: true,
                    checkedQuantity: allocation.quantity,
                    checkedAt: moment().format(),
                    checkedUser: userId,
                    checked: true,
                  }
                : {}),
            },
            {
              transaction,
            }
          );
        } else {
          const productRange = {
            outboundOrderId: outboundOrder.id,
            outboundOrderProductId: allocation.outboundOrderProductId,
            outboundRangeId: outboundOrderRange.id,
            allocationId: allocation.id,
            productCode: stockBalance.productCode,
            quantity: allocation.quantity,
            status,
            cuttingPlan: status === 1,
            note:
              outboundOrder.typeOrders &&
              outboundOrder.typeOrders.usesRequestManual
                ? outboundOrder.note
                : '',
            createdUser: userId,
            updatedUser: userId,
            ...(industry
              ? {
                  storageAddressPicked: stockBalance.addressCode,
                  pickedQuantity: allocation.quantity,
                  pickedAt: moment().format(),
                  pickedUser: userId,
                  picked: true,
                  checkedQuantity: allocation.quantity,
                  checkedAt: moment().format(),
                  checkedUser: userId,
                  checked: true,
                }
              : {}),
          };
 
          await OutboundRangeCreate.createOutboundRangeProduct(productRange, {
            transaction,
          });
        }
      }
    }
  } catch (error) {
    // eslint-disable-next-line no-console
    /* eslint-disable */console.log(...oo_oo(`3444396944_298_4_298_22_4`,error));
    throw error;
  }
}
 
async function deleteOutboundRanges(outboundProduct, transaction) {
  try {
    const outboundProductRanges =
      await OutboundRangeInformation.getOutboundRangeProducts(
        {
          outboundOrderProductId: outboundProduct.id,
          status: [1, 2],
        },
        { transaction }
      );
 
    if (outboundProductRanges && outboundProductRanges.length > 0) {
      for (const productRange of outboundProductRanges) {
        await OutboundRangeDelete.deleteOutboundRangeProduct(productRange.id, {
          transaction,
        });
 
        const { outboundRangeId } = productRange;
 
        const productsRange =
          await OutboundRangeInformation.getOutboundRangeProducts(
            {
              outboundRangeId,
            },
            { transaction }
          );
 
        if (!productsRange || productsRange.length === 0) {
          await OutboundRangeDelete.deleteOutboundRange(outboundRangeId, {
            transaction,
          });
        }
      }
    }
  } catch (error) {
    // eslint-disable-next-line no-console
    /* eslint-disable */console.log(...oo_oo(`3444396944_339_4_339_22_4`,error));
  }
}

  outboundRangeProductId,
  quantityToIncrement,
  userId,
  transaction
) {

        outboundRangeProductId,

          picked: false,
